# Die Wacht am Rhein
flavor_RHLND.2 = {
	type = country_event
	title = flavor_RHLND.2.title
	desc = flavor_RHLND.2.desc
	fire_only_once = yes

	# TODO: check that this works
	trigger = { # Triggers after 120 months of war during age of absolutism
		AND = {
			tag = RHLND
			current_age_or_later = { age = age_5_absolutism }
			months_since_war > 120
		}
	}

	immediate = {
		# sets the required variables to meet the rheinwachtgesetz recruitment law, once NOT_MET has expired you will be able to pass the law
		set_variable = { 
			name = fnr_has_die_wacht_am_rhein_10years_NOT_MET
			value = yes
			days = 3650
		}
		set_variable = { 
			name = fnr_has_die_wacht_am_rhein_10years_HAS_MET
			value = yes
			days = -1
		}
	}

	option = { # Das Rheinische Königreich
		name = flavor_RHLND.2.a
		add_reform = government_reform:fnr_GSCTL_die_wacht_am_rhein
	}
}

# The Pillar that is the Church
# Event to replace High Church Influence with The Pillar that is the Church 
# (Must not have manually revoked the reform, if so, the event will fire but instead provide the Clergy with 10,000 gold )
# Manually removed variable: fnr_high_church_influence_HAS_removed
# Removed via event variable: fnr_high_church_influence_removed
flavor_RHLND.4 = {
	type = country_event
	title = flavor_RHLND.4.title
	desc = flavor_RHLND.4.desc
	fire_only_once = yes

	trigger = {
		tag = RHLND
		NOT = { has_estate_privilege = estate_privilege:fnr_RHLND_pillar_that_is_the_church }
		OR = {
			AND = { # Manually removed check 
					# Check that either the high church influence privilege is still enacted
				has_variable = fnr_high_church_influence_HAS_removed
				has_estate_privilege = estate_privilege:fnr_RHLND_high_church_influence 
				any_estate = {
					estate_type = estate_type:clergy_estate
					power < 0.10
				}
				state_religion_clergy < 0.10
			}
			AND = { # Event fired check
					# fnr_RHLND_high_church_influence has been removed and the nation has less than 10 clergy power or 10% clergy pops
					# TODO POST 2.5.0: fix this event, it should have different triggers, for now NOT = { has_estate_privilege = estate_privilege:fnr_RHLND_pillar_that_is_the_church } is checked regardless.
				NOT = { has_variable = fnr_high_church_influence_HAS_removed }
				NOT = { has_estate_privilege = estate_privilege:fnr_RHLND_pillar_that_is_the_church }
				NOT = { has_fired_unique_event = flavor_RHLND.4 }
				any_estate = {
					estate_type = estate_type:clergy_estate
					power < 0.10
				}
				state_religion_clergy < 0.10
			}
		}
	}

	option = {
		name = flavor_RHLND.4.a
		set_variable = fnr_high_church_influence_removed
		revoke_estate_privilege = estate_privilege:fnr_RHLND_high_church_influence 
		grant_estate_privilege = estate_privilege:fnr_RHLND_pillar_that_is_the_church
		# Make a check to see if manually removed, or event removed, if manual removed instead of replacing the privilege it should give stability, but also clergy power
	}
}

# State of the Confederation - Government Rank Handler - Works in comparison with New Path
# Allows you to stay as a rank 1, but at a cost
flavor_RHLND.5 = {
	type = country_event
	title = flavor_RHLND.5.title
	desc = flavor_RHLND.5.desc
	fire_only_once = yes

	option = {
		name = flavor_RHLND.5.a
		if = {
			limit = { country_rank_level < 1 }
			set_country_rank_effect = { rank = country_rank:rank_county }
		}
		scope:recipient = {	
			add_country_modifier = {
				modifier = fnr_RHLND_limitations_of_united_counties
				years = 5
				mode = add_and_extend
			}
		}
		add_stability = -5
	}

	option = {
		name = flavor_RHLND.5.b
		if = {
			limit = { country_rank_level < 2 }
			set_country_rank_effect = { rank = country_rank:rank_duchy }
		}
	}
}
