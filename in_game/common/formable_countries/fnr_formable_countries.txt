						text = NED_f_tt2
					}
					if = { # Option 1: Decentralized (-6 or lower) → United Provinces
						limit = {
							AND = {
								has_societal_value = societal_value_type:centralization_vs_decentralization
								societal_value:centralization_vs_decentralization > -5
							}
						}
						# Become United Provinces
						change_country_name = NED_UP
						set_variable = NED_UP_var # Set variable allowing future switch to Dutch Republic
						hidden_effect = { change_country_flag = NED_UP_FLAG }
					}
					else_if = { # Option 2: Centralized (+6 or higher) → Dutch Republic
						limit = {
							AND = {
								has_societal_value = societal_value_type:centralization_vs_decentralization
								societal_value:centralization_vs_decentralization < 5
							}
						}
						# Become Dutch Republic
						change_country_name = NED_DR
						set_variable = NED_DR_var
						hidden_effect = { change_country_flag = NED_DR_FLAG }
						# Do NOT set variable - cannot switch to United Provinces
					}
					else = { # Option 3: Balanced (-5 to +5) → Netherlands
						limit = {
							AND = {
								has_societal_value = societal_value_type:centralization_vs_decentralization
								societal_value:centralization_vs_decentralization < -6
								has_societal_value = societal_value_type:centralization_vs_decentralization
								societal_value:centralization_vs_decentralization > 6
							}
						}
						add_prestige = 5
						# Remain as Netherlands (NED)
						# No cosmetic tag change needed
					}
				}
