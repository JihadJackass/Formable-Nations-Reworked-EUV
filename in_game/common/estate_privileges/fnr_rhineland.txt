	# Clergy

		# High Church Influence
		fnr_RHLND_high_church_influence = {
			estate = clergy_estate
			
			potential = {
				OR = {
					tag = RHLND
					has_or_had_tag = RHLND
				}
			}
			allow = {
				OR = {
					tag = RHLND
					has_or_had_tag = RHLND
				}
				NOT = { has_variable = fnr_high_church_influence_removed }
			}
			
			country_modifier = {
				clergy_estate_can_participate_in_parliament = yes
				cultural_tradition_modifier = 0.05
				nobles_estate_max_tax = 0.05
				blocked_from_conversion = yes

				global_clergy_estate_power = 0.50
				clergy_estate_target_satisfaction = -0.05

				monthly_towards_spiritualist = 0.25
			}

			on_deactivate = {
				# Check that the Church Solutions event has not fired
				set_variable = fnr_high_church_influence_HAS_removed
				remove_variable = fnr_high_church_influence_NOT_removed
				revoke_estate_privilege = estate_privilege:fnr_RHLND_high_church_influence
				grant_estate_privilege = estate_privilege:fnr_RHLND_pillar_that_is_the_church
			}
			
			can_revoke = {
				# Must have at least less than 10 clergy estate power AND less than 10% clergy pop, or the Church Solutions event must have fired (50-100 year timer)
				AND = {
					any_estate = {
						estate_type = estate_type:clergy_estate
						power < 0.15
					}
					state_religion_clergy < 0.10
				}
			}
		}
