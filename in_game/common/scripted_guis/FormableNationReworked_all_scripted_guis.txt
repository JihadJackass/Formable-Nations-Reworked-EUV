# Initialize/Toggle Debug Window
FormableNationsReworkedDebug_initialize = {
	scope = country
	saved_scopes = { }
	is_shown = { always = yes }
	is_valid = { always = yes }
	effect = {
		# This scripted GUI is called when opening the debug window
		# Currently the window is controlled by the Toggle variable, so this can be empty
		# Future: Could initialize default settings or perform setup here
	}
}

# Toggle for Historical Formables - fnr_historical_enabled
	# Existed after EU5 start date and not after January 1, 1837
	FormableNationReworked_toggle_historical_formables = {
		scope = country
		is_shown = { always = yes }
		is_valid = { always = yes }
		effect = {
			# Check player's current state
			if = {
				limit = { has_variable = fnr_historical_enabled }
				# Currently enabled, so disable for ALL countries
				every_country = {
					remove_variable = fnr_historical_enabled
				}
			}
			else = {
				# Currently disabled, so enable for ALL countries
				every_country = {
					set_variable = {
						name = fnr_historical_enabled
						value = yes
					}
				}
			}
		}
	}

			FormableNationReworked_toggle_arpitania = {
				scope = country
				is_shown = { always = yes }
				is_valid = { always = yes }
				effect = {
					if = { # Cycle: Default → Force OFF → Force ON → Default
						limit = {
							NOT = { has_variable = fnr_arpitania_enabled }
							NOT = { has_variable = fnr_arpitania_disabled }
						}
						every_country = { # Currently DEFAULT → Force OFF
							set_variable = { name = fnr_arpitania_disabled value = yes }
						}
					}
					else_if = { # Currently FORCE OFF → Force ON
						limit = { has_variable = fnr_arpitania_disabled }
						every_country = { 
							remove_variable = fnr_arpitania_disabled
							set_variable = { name = fnr_arpitania_enabled value = yes }
						}
					}
					else_if = { # Currently FORCE ON → Default
						limit = { has_variable = fnr_arpitania_enabled }
						every_country = {
							remove_variable = fnr_arpitania_enabled
						}
					}
				}
			}
